Actualizar registros

Ejemplos

Actualizar un registro


            UPDATE Estudiantes
            SET nombre = 'Juan Pérez'
            WHERE idEstudiante = 1;

Actualizar varias columnas


            UPDATE Estudiantes
            SET nombre = 'Carlos Ruiz',
                edad = 21
            WHERE idEstudiante = 3;

Actualizar toda la tabla


            UPDATE Estudiantes
            SET edad = 20;

Actualizar con condicion


            UPDATE empleados
            SET salario = salario + 1000
            WHERE departamento = 'Ventas';

Actualizar con operadores matematicos


            UPDATE productos
            SET stock = stock + 10
            WHERE id = 5;

Actualizar entre rangos


            UPDATE empleados
            SET bono = 1000
            WHERE salario BETWEEN 3000 AND 5000;

Actualizar usando LIKE


            UPDATE clientes
            SET categoria = 'Premium'
            WHERE nombre LIKE 'Juan%';
