Seleccionar registros

Ejemplos

Seleccionar todos los registros


            SELECT *
            FROM empleados;

Seleccionar columnas especificas


            SELECT nombre, salario, departamento
            FROM empleados;

Seleccionar un registro especifico con WHERE


            SELECT *
            FROM empleados
            WHERE id = 1;

Seleccionar un registro con operadores de comparacion


            SELECT *
            FROM empleados
            WHERE salario > 5000;

Seleccionar usando AND


            SELECT *
            FROM empleados
            WHERE departamento = 'Ventas'
            AND salario > 5000;

Seleccionar usando OR


            SELECT *
            FROM empleados
            WHERE departamento = 'Ventas'
            OR departamento = 'Marketing';

Seleccionar usando NOT


            SELECT *
            FROM empleados
            WHERE NOT departamento = 'Ventas';

Seleccionar por rangos


            SELECT *
            FROM empleados
            WHERE salario BETWEEN 3000 AND 6000;

Seleccionar valores especificos


            SELECT *
            FROM empleados
            WHERE departamento IN ('Ventas', 'Marketing', 'RH');

Seleccionar con patrones


            SELECT *
            FROM empleados
            WHERE nombre LIKE 'Juan%';
