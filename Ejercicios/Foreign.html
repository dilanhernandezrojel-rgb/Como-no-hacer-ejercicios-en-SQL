Foreign key/Normalizacion

Ejemplos

Tabla con clave primaria


            CREATE TABLE Estudiantes (
                idEstudiante INT PRIMARY KEY,
                nombre VARCHAR(100),
                edad INT
            );

El identificador se genera automaticamente


            CREATE TABLE Estudiantes (
                idEstudiante INT AUTO_INCREMENT PRIMARY KEY,
                nombre VARCHAR(100),
                edad INT
            );

Tabla principal


            CREATE TABLE Carreras (
                idCarrera INT PRIMARY KEY,
                nombreCarrera VARCHAR(100)
            );

Tabla con clave foránea


            CREATE TABLE Estudiantes (
                idEstudiante INT PRIMARY KEY,
                nombre VARCHAR(100),
                idCarrera INT,
    
                FOREIGN KEY (idCarrera)
                REFERENCES Carreras(idCarrera)
            );
